## üîó Oracle Agent Spec Implementation

### Python Implementation Template

```python
from pyagentspec.agent import Agent
from pyagentspec.llms import VllmConfig
from pyagentspec.property import StringProperty, NumberProperty, ObjectProperty, ListProperty
from pyagentspec.tools import ServerTool
from pyagentspec.serialization import AgentSpecSerializer


def create_[agent_name]_agent():
    """
    [Agent description and purpose]
    ROI: [efficiency gain] | $[annual savings]
    """

    llm_config = VllmConfig(
        name="[agent-name]-llm",
        model_id="[model-id]",
        url="[your-llm-endpoint-url]"
    )

    tools = [
        ServerTool(
            name="[tool_name]",
            description="[Clear description of what this tool does]",
            inputs=[
                StringProperty(title="[param]", description="[description]"),
                # Add more input properties
            ],
            outputs=[
                ObjectProperty(
                    title="[result_name]",
                    properties=[
                        StringProperty(title="[field]"),
                        # Add more output properties
                    ]
                )
            ]
        ),
        # Add more tools
    ]

    agent = Agent(
        name="[Agent Display Name]",
        llm_config=llm_config,
        system_prompt="""[System prompt from section above]""",
        tools=tools,
        inputs=[
            StringProperty(title="user_name", description="User's name"),
            StringProperty(title="department", description="User's department"),
            StringProperty(title="access_level", description="User access level")
            # Add context variables
        ]
    )

    return agent


def main():
    """Generate and save the agent configuration"""
    agent = create_[agent_name]_agent()

    serializer = AgentSpecSerializer()

    # Save as JSON
    agent_json = serializer.to_json(agent)
    with open("configs/[agent_name].json", "w") as f:
        f.write(agent_json)

    # Save as YAML
    agent_yaml = serializer.to_yaml(agent)
    with open("configs/[agent_name].yaml", "w") as f:
        f.write(agent_yaml)

    print(f"‚úì [Agent Name] created successfully")
    print(f"  ROI: [X]% efficiency gain | $[Y]K annual savings")
    print(f"  Configurations saved to configs/")


if __name__ == "__main__":
    main()
```
